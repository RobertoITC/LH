#include 
#include <string>

// Function to add two numbers and return the result</span>
int add(int num1, int num2) {
    return num1 + num2;
}

int main() {
    int a, b;
    std::string response;

    // Ask user for two numbers</span>
    std::cout << string'>"Enter the first number: "</span>;
    std::cin >> a;
    std::cout << string'>"Enter the second number: "</span>;
    std::cin >> b;

    // Perform addition using the add function</span>
    int result = add(a, b);

    // Create a response string and output it</span>
    response = string'>"The sum of "</span> + std::to_string(a) + string'>" and "</span> + std::to_string(b) + string'>" is "</span> + std::to_string(result) + string'>"."</span>;
    std::cout << response << std::endl;

    return 0;
}
